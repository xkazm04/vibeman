**direction.dart:**
Create Direction data model:
```dart
class Direction {
  final String id;
  final String projectId;
  final String projectName;
  final String summary;
  final String? direction; // Full markdown content
  final String? contextName;
  final DateTime createdAt;

  Direction({
    required this.id,
    required this.projectId,
    required this.projectName,
    required this.summary,
    this.direction,
    this.contextName,
    required this.createdAt,
  });

  factory Direction.fromSupabase(Map<String, dynamic> json) {
    final payload = json['payload'] as Map<String, dynamic>? ?? {};
    return Direction(
      id: payload['direction_id'] as String? ?? json['id'] as String,
      projectId: json['project_id'] as String,
      projectName: json['project_name'] as String? ?? 'Unknown',
      summary: payload['summary'] as String? ?? 'No summary',
      direction: payload['direction'] as String?,
      contextName: payload['context_name'] as String?,
      createdAt: DateTime.parse(json['created_at'] as String),
    );
  }
}
```
