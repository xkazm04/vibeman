/// Requirement model for batch execution
class Requirement {
  final String id;
  final String name;
  final String projectId;
  final String projectName;
  final String? content;
  final DateTime syncedAt;

  Requirement({
    required this.id,
    required this.name,
    required this.projectId,
    required this.projectName,
    this.content,
    required this.syncedAt,
  });

  factory Requirement.fromSupabase(Map<String, dynamic> json) {
    final payload = json['payload'] as Map<String, dynamic>? ?? {};
    return Requirement(
      id: json['id']?.toString() ?? '',
      name: payload['requirement_name'] as String? ?? '',
      projectId: json['project_id'] as String? ?? '',
      projectName: json['project_name'] as String? ?? '',
      content: payload['content'] as String?,
      syncedAt: DateTime.parse(json['created_at'] as String),
    );
  }
}

import 'package:flutter/foundation.dart';
import 'package:supabase_flutter/supabase_flutter.dart';
import '../models/requirement.dart';

/// Healthcheck state from Vibeman
class HealthcheckState {
  final bool zenMode;
  final int activeSessions;
  final int availableSlots;
  final DateTime timestamp;

  HealthcheckState({
    required this.zenMode,
    required this.activeSessions,
    required this.availableSlots,
    required this.timestamp,
  });
}

/// Execution status for a batch
enum ExecutionStatus { idle, running, completed, failed }

class BatchService extends ChangeNotifier {
  final SupabaseClient _supabase;

  HealthcheckState? _healthcheck;
  RealtimeChannel? _healthcheckChannel;
  RealtimeChannel? _executionChannel;

  List<Requirement> _requirements = [];
  Set<String> _selectedIds = {};
  ExecutionStatus _executionStatus = ExecutionStatus.idle;
  String? _currentSessionId;
  int _completedCount = 0;
  int _failedCount = 0;
  String? _lastError;

  BatchService(this._supabase);

  // Getters
  HealthcheckState? get healthcheck => _healthcheck;
  List<Requirement> get requirements => _requirements;
  Set<String> get selectedIds => _selectedIds;
  ExecutionStatus get executionStatus => _executionStatus;
  String? get currentSessionId => _currentSessionId;
  int get completedCount => _completedCount;
  int get failedCount => _failedCount;
  String? get lastError => _lastError;

  bool get canStartBatch =>
      _healthcheck?.zenMode == true &&
      (_healthcheck?.availableSlots ?? 0) > 0 &&
      _selectedIds.isNotEmpty &&
      _executionStatus != ExecutionStatus.running;

  int get availableSlots => _healthcheck?.availableSlots ?? 0;

  /// Load requirements from Supabase for a project
  Future<void> loadRequirements(String projectId) async {
    try {
      final response = await _supabase
          .from('vibeman_events')
          .select()
          .eq('project_id', projectId)
          .eq('event_type', 'requirement_pending')
          .order('created_at', ascending: false);

      _requirements = (response as List)
          .map((json) => Requirement.fromSupabase(json))
          .toList();

      _selectedIds.clear();
      notifyListeners();
    } catch (e) {
      debugPrint('Error loading requirements: $e');
    }
  }

  /// Toggle requirement selection
  void toggleSelection(String requirementId) {
    if (_selectedIds.contains(requirementId)) {
      _selectedIds.remove(requirementId);
    } else {
      _selectedIds.add(requirementId);
    }
    notifyListeners();
  }

  /// Select all requirements
  void selectAll() {
    _selectedIds = _requirements.map((r) => r.id).toSet();
    notifyListeners();
  }

  /// Clear selection
  void clearSelection() {
    _selectedIds.clear();
    notifyListeners();
  }

  /// Subscribe to healthcheck updates for a project
  void subscribeToHealthcheck(String projectId) {
    _healthcheckChannel?.unsubscribe();

    _healthcheckChannel = _supabase
        .channel('healthcheck:$projectId')
        .onPostgresChanges(
          event: PostgresChangeEvent.insert,
          schema: 'public',
          table: 'vibeman_events',
          filter: PostgresChangeFilter(
            type: PostgresChangeFilterType.eq,
            column: 'event_type',
            value: 'healthcheck',
          ),
          callback: (payload) {
            final data = payload.newRecord['payload'] as Map<String, dynamic>?;
            if (data != null) {
              _healthcheck = HealthcheckState(
                zenMode: data['zen_mode'] as bool? ?? false,
                activeSessions: data['active_sessions'] as int? ?? 0,
                availableSlots: data['available_slots'] as int? ?? 4,
                timestamp: DateTime.parse(
                    data['timestamp'] as String? ?? DateTime.now().toIso8601String()),
              );
              notifyListeners();
            }
          },
        )
        .subscribe();
  }

  /// Subscribe to execution events for monitoring
  void subscribeToExecution(String projectId) {
    _executionChannel?.unsubscribe();

    _executionChannel = _supabase
        .channel('execution:$projectId')
        .onPostgresChanges(
          event: PostgresChangeEvent.insert,
          schema: 'public',
          table: 'vibeman_events',
          callback: (payload) {
            final eventType = payload.newRecord['event_type'] as String?;
            final data = payload.newRecord['payload'] as Map<String, dynamic>?;

            if (eventType == 'task_completed') {
              _completedCount++;
              notifyListeners();
            } else if (eventType == 'task_failed') {
              _failedCount++;
              _lastError = data?['error'] as String?;
              notifyListeners();
            }
          },
        )
        .subscribe();
  }

  /// Start batch execution
  Future<bool> startBatch(String projectId) async {
    if (!canStartBatch) return false;

    final selectedRequirements = _requirements
        .where((r) => _selectedIds.contains(r.id))
        .toList();

    if (selectedRequirements.isEmpty) return false;

    _executionStatus = ExecutionStatus.running;
    _completedCount = 0;
    _failedCount = 0;
    _lastError = null;
    notifyListeners();

    try {
      await _supabase.from('vibeman_commands').insert({
        'project_id': projectId,
        'command_type': 'start_batch',
        'payload': {
          'requirement_names': selectedRequirements.map((r) => r.name).toList(),
          'project_id': projectId,
        },
        'status': 'pending',
      });

      return true;
    } catch (e) {
      debugPrint('Error starting batch: $e');
      _executionStatus = ExecutionStatus.failed;
      _lastError = e.toString();
      notifyListeners();
      return false;
    }
  }

  /// Reset execution state
  void resetExecution() {
    _executionStatus = ExecutionStatus.idle;
    _currentSessionId = null;
    _completedCount = 0;
    _failedCount = 0;
    _lastError = null;
    notifyListeners();
  }

  @override
  void dispose() {
    _healthcheckChannel?.unsubscribe();
    _executionChannel?.unsubscribe();
    super.dispose();
  }
}
