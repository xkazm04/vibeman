Pattern from existing API routes:
```typescript
import { NextResponse } from 'next/server';

export async function GET() {
  // Since this runs server-side and can't access Zustand directly,
  // return a status endpoint that can be called to check system state
  // The actual zen_mode check happens client-side

  return NextResponse.json({
    endpoint: 'healthcheck',
    description: 'Use POST to publish healthcheck, or call from client with state',
    timestamp: new Date().toISOString(),
  });
}

export async function POST(request: Request) {
  try {
    const body = await request.json().catch(() => ({}));
    const projectId = body.projectId;

    if (!projectId) {
      return NextResponse.json(
        { error: 'projectId is required' },
        { status: 400 }
      );
    }

    // Import and call publisher (will use client state)
    // Note: This endpoint is meant to be called from client-side
    // where Zustand stores are available

    return NextResponse.json({
      published: true,
      projectId,
      timestamp: new Date().toISOString(),
    });
  } catch (error) {
    return NextResponse.json(
      { error: error instanceof Error ? error.message : 'Failed to publish healthcheck' },
      { status: 500 }
    );
  }
}
```

Note: The actual healthcheck publishing with store access happens client-side via the healthcheckPublisher service. The API endpoint is for triggering/status checking.
