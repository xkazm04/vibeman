import 'package:flutter/foundation.dart';
import 'package:flutter_secure_storage/flutter_secure_storage.dart';

/// Connection status for Supabase
enum SupabaseConnectionStatus {
  /// Not configured - no credentials saved
  notConfigured,
  /// Connecting to Supabase
  connecting,
  /// Connected and receiving realtime events
  connected,
  /// Disconnected - connection lost
  disconnected,
  /// Error state - invalid credentials or network issue
  error,
}

/// Project information from Supabase
class VibemanProject {
  final String id;
  final String name;
  final DateTime? lastSyncAt;

  VibemanProject({
    required this.id,
    required this.name,
    this.lastSyncAt,
  });

  factory VibemanProject.fromJson(Map<String, dynamic> json) {
    return VibemanProject(
      id: json['id'] as String,
      name: json['name'] as String? ?? json['project_name'] as String? ?? 'Unknown',
      lastSyncAt: json['last_sync_at'] != null
          ? DateTime.tryParse(json['last_sync_at'] as String)
          : null,
    );
  }
}

/// Service for managing Supabase connection
class SupabaseConnectionService extends ChangeNotifier {
  static const _urlKey = 'supabase_url';
  static const _anonKeyKey = 'supabase_anon_key';
  static const _selectedProjectKey = 'selected_project_id';

  final _secureStorage = const FlutterSecureStorage();

  SupabaseConnectionStatus _status = SupabaseConnectionStatus.notConfigured;
  String? _errorMessage;
  List<VibemanProject> _projects = [];
  String? _selectedProjectId;
  RealtimeChannel? _channel;
  bool _initialized = false;

  // Getters
  SupabaseConnectionStatus get status => _status;
  String? get errorMessage => _errorMessage;
  List<VibemanProject> get projects => _projects;
  String? get selectedProjectId => _selectedProjectId;
  VibemanProject? get selectedProject =>
      _projects.where((p) => p.id == _selectedProjectId).firstOrNull;
  bool get isConfigured => _status != SupabaseConnectionStatus.notConfigured;
  bool get isConnected => _status == SupabaseConnectionStatus.connected;

  /// Map connection status to StatusDot Status enum
  Status get statusDotStatus {
    switch (_status) {
      case SupabaseConnectionStatus.notConfigured:
        return Status.inactive;
      case SupabaseConnectionStatus.connecting:
        return Status.warning;
      case SupabaseConnectionStatus.connected:
        return Status.active;
      case SupabaseConnectionStatus.disconnected:
      case SupabaseConnectionStatus.error:
        return Status.error;
    }
  }

  /// Initialize the service
  Future<void> initialize() async {
    if (_initialized) return;
    _initialized = true;

    // Load saved credentials and try to connect
    final url = await _secureStorage.read(key: _urlKey);
    final anonKey = await _secureStorage.read(key: _anonKeyKey);

    if (url != null && anonKey != null && url.isNotEmpty && anonKey.isNotEmpty) {
      await _connectWithCredentials(url, anonKey);
    }

    // Load last selected project
    final prefs = await SharedPreferences.getInstance();
    _selectedProjectId = prefs.getString(_selectedProjectKey);
    notifyListeners();
  }

  /// Save credentials and connect
  Future<bool> saveCredentials({
    required String url,
    required String anonKey,
  }) async {
    try {
      await _secureStorage.write(key: _urlKey, value: url);
      await _secureStorage.write(key: _anonKeyKey, value: anonKey);

      return await _connectWithCredentials(url, anonKey);
    } catch (e) {
      _status = SupabaseConnectionStatus.error;
      _errorMessage = 'Failed to save credentials: $e';
      notifyListeners();
      return false;
    }
  }

  /// Connect with given credentials
  Future<bool> _connectWithCredentials(String url, String anonKey) async {
    _status = SupabaseConnectionStatus.connecting;
    _errorMessage = null;
    notifyListeners();

    try {
      // Initialize Supabase
      await Supabase.initialize(
        url: url,
        anonKey: anonKey,
      );

      // Subscribe to a test channel to verify realtime works
      _setupRealtimeSubscription();

      // Fetch projects
      await _fetchProjects();

      _status = SupabaseConnectionStatus.connected;
      notifyListeners();
      return true;
    } catch (e) {
      _status = SupabaseConnectionStatus.error;
      _errorMessage = e.toString();
      notifyListeners();
      return false;
    }
  }

  /// Setup realtime subscription for connection monitoring
  void _setupRealtimeSubscription() {
    _channel?.unsubscribe();

    final client = Supabase.instance.client;
    _channel = client.channel('connection_monitor');

    _channel!.subscribe((status, [error]) {
      if (status == RealtimeSubscribeStatus.subscribed) {
        _status = SupabaseConnectionStatus.connected;
        _errorMessage = null;
      } else if (status == RealtimeSubscribeStatus.closed) {
        _status = SupabaseConnectionStatus.disconnected;
      } else if (status == RealtimeSubscribeStatus.channelError) {
        _status = SupabaseConnectionStatus.error;
        _errorMessage = error?.toString();
      }
      notifyListeners();
    });
  }

  /// Fetch projects from Supabase
  Future<void> _fetchProjects() async {
    try {
      final client = Supabase.instance.client;
      // Projects are stored in vibeman_events with unique project_ids
      // Or we can query a dedicated projects table if it exists
      final response = await client
          .from('vibeman_events')
          .select('project_id, project_name')
          .limit(100);

      // Deduplicate by project_id
      final Map<String, VibemanProject> projectMap = {};
      for (final row in response as List) {
        final id = row['project_id'] as String?;
        if (id != null && !projectMap.containsKey(id)) {
          projectMap[id] = VibemanProject(
            id: id,
            name: row['project_name'] as String? ?? id,
          );
        }
      }

      _projects = projectMap.values.toList();

      // Auto-select first project if none selected
      if (_selectedProjectId == null && _projects.isNotEmpty) {
        _selectedProjectId = _projects.first.id;
        final prefs = await SharedPreferences.getInstance();
        await prefs.setString(_selectedProjectKey, _selectedProjectId!);
      }
    } catch (e) {
      // Projects fetch failed - not critical, continue
      debugPrint('Failed to fetch projects: $e');
    }
  }

  /// Select a project
  Future<void> selectProject(String projectId) async {
    _selectedProjectId = projectId;
    final prefs = await SharedPreferences.getInstance();
    await prefs.setString(_selectedProjectKey, projectId);
    notifyListeners();
  }

  /// Clear credentials and disconnect
  Future<void> disconnect() async {
    await _channel?.unsubscribe();
    _channel = null;
    await _secureStorage.delete(key: _urlKey);
    await _secureStorage.delete(key: _anonKeyKey);
    _status = SupabaseConnectionStatus.notConfigured;
    _projects = [];
    _selectedProjectId = null;
    notifyListeners();
  }

  /// Refresh projects list
  Future<void> refreshProjects() async {
    if (isConnected) {
      await _fetchProjects();
      notifyListeners();
    }
  }

  @override
  void dispose() {
    _channel?.unsubscribe();
    super.dispose();
  }
}

import 'package:flutter/material.dart';

/// Widget showing Supabase connection status and project selector
class SupabaseStatusIndicator extends StatelessWidget {
  final bool showProjectDropdown;
  final bool isCompact;

  const SupabaseStatusIndicator({
    super.key,
    this.showProjectDropdown = true,
    this.isCompact = false,
  });

  @override
  Widget build(BuildContext context) {
    return ListenableBuilder(
      listenable: locator<SupabaseConnectionService>(),
      builder: (context, _) {
        final service = locator<SupabaseConnectionService>();

        return Row(
          mainAxisSize: MainAxisSize.min,
          children: [
            // Status dot
            Tooltip(
              message: _getTooltipMessage(service),
              child: StatusDot(
                status: service.statusDotStatus,
                size: isCompact ? 6 : 8,
                showGlow: true,
              ),
            ),

            // Project dropdown (when connected and enabled)
            if (showProjectDropdown &&
                service.isConnected &&
                service.projects.isNotEmpty) ...[
              const SizedBox(width: 8),
              _ProjectDropdown(
                service: service,
                isCompact: isCompact,
              ),
            ],
          ],
        );
      },
    );
  }

  String _getTooltipMessage(SupabaseConnectionService service) {
    switch (service.status) {
      case SupabaseConnectionStatus.notConfigured:
        return 'Supabase not configured';
      case SupabaseConnectionStatus.connecting:
        return 'Connecting to Supabase...';
      case SupabaseConnectionStatus.connected:
        return 'Connected to Supabase';
      case SupabaseConnectionStatus.disconnected:
        return 'Disconnected from Supabase';
      case SupabaseConnectionStatus.error:
        return 'Supabase error: ${service.errorMessage ?? "Unknown"}';
    }
  }
}

class _ProjectDropdown extends StatelessWidget {
  final SupabaseConnectionService service;
  final bool isCompact;

  const _ProjectDropdown({
    required this.service,
    required this.isCompact,
  });

  @override
  Widget build(BuildContext context) {
    final selectedProject = service.selectedProject;

    return PopupMenuButton<String>(
      tooltip: 'Select project',
      initialValue: service.selectedProjectId,
      onSelected: service.selectProject,
      offset: const Offset(0, 40),
      itemBuilder: (context) {
        return service.projects.map((project) {
          return PopupMenuItem<String>(
            value: project.id,
            child: Row(
              children: [
                if (project.id == service.selectedProjectId)
                  Icon(
                    Icons.check,
                    size: 16,
                    color: Theme.of(context).colorScheme.primary,

                else
                  const SizedBox(width: 16),
                const SizedBox(width: 8),
                Expanded(
                  child: Text(
                    project.name,

              ],
            ),
          );
        }).toList();
      },
      child: Container(
        padding: const EdgeInsets.symmetric(horizontal: 8, vertical: 4),

          color: Colors.white.withOpacity(0.1),
          borderRadius: BorderRadius.circular(4),
        ),

          mainAxisSize: MainAxisSize.min,

            Text(
              selectedProject?.name ?? 'Select',
              style: TextStyle(
                fontFamily: isCompact ? null : 'RobotoMono',

              Icons.arrow_drop_down,
              size: 16,
              color: Colors.white.withOpacity(0.6),
            ),
          ],
        ),
      ),
    );
  }
}

/// Configuration section for Supabase connection settings
class SupabaseConfigSection extends StatefulWidget {
  final Color? accentColor;
  final TextStyle? labelStyle;
  final InputDecoration? inputDecoration;

  const SupabaseConfigSection({
    super.key,
    this.accentColor,
    this.labelStyle,
    this.inputDecoration,
  });

  @override
  State<SupabaseConfigSection> createState() => _SupabaseConfigSectionState();
}

class _SupabaseConfigSectionState extends State<SupabaseConfigSection> {
  final _urlController = TextEditingController();
  final _anonKeyController = TextEditingController();
  bool _isSaving = false;
  String? _error;

  SupabaseConnectionService get _service => locator<SupabaseConnectionService>();

  @override
  void initState() {
    super.initState();
    // Pre-fill if already configured (would need secure storage read)
    // For now, leave empty for security
  }

  @override
  void dispose() {
    _urlController.dispose();
    _anonKeyController.dispose();
    super.dispose();
  }

  Future<void> _saveCredentials() async {
    final url = _urlController.text.trim();
    final anonKey = _anonKeyController.text.trim();

    if (url.isEmpty || anonKey.isEmpty) {
      setState(() => _error = 'Both URL and Anon Key are required');
      return;
    }

    if (!url.startsWith('https://') || !url.contains('supabase')) {
      setState(() => _error = 'Invalid Supabase URL');
      return;
    }

    setState(() {
      _isSaving = true;
      _error = null;
    });

    final success = await _service.saveCredentials(
      url: url,
      anonKey: anonKey,
    );

    setState(() {
      _isSaving = false;
      if (!success) {
        _error = _service.errorMessage ?? 'Failed to connect';
      }
    });
  }

  Future<void> _disconnect() async {
    await _service.disconnect();
    _urlController.clear();
    _anonKeyController.clear();
    setState(() {});
  }

  @override
  Widget build(BuildContext context) {
    return ListenableBuilder(
      listenable: _service,
      builder: (context, _) {
        final isConnected = _service.isConnected;
        final isConfigured = _service.isConfigured;

        return Column(
          crossAxisAlignment: CrossAxisAlignment.start,
          children: [
            // Header with status
            Row(
              children: [
                Text(
                  'SUPABASE CONNECTION',
                  style: widget.labelStyle ?? const TextStyle(
                    fontWeight: FontWeight.bold,
                    letterSpacing: 1,
                  ),
                ),
                const Spacer(),
                StatusDot(status: _service.statusDotStatus),
                const SizedBox(width: 8),
                Text(
                  _getStatusText(),
                  style: TextStyle(
                    fontSize: 12,
                    color: Colors.grey[400],
                  ),
                ),
              ],
            ),
            const SizedBox(height: 16),

            // If connected, show disconnect option
            if (isConnected) ...[
              Container(
                padding: const EdgeInsets.all(12),
                decoration: BoxDecoration(
                  color: Colors.green.withOpacity(0.1),
                  borderRadius: BorderRadius.circular(8),
                  border: Border.all(color: Colors.green.withOpacity(0.3)),
                ),
                child: Row(
                  children: [
                    const Icon(Icons.check_circle, color: Colors.green, size: 20),
                    const SizedBox(width: 12),
                    Expanded(
                      child: Column(
                        crossAxisAlignment: CrossAxisAlignment.start,
                        children: [
                          const Text(
                            'Connected',
                            style: TextStyle(
                              fontWeight: FontWeight.bold,
                              color: Colors.green,
                            ),
                          ),
                          Text(
                            '${_service.projects.length} project(s) available',
                            style: TextStyle(
                              fontSize: 12,
                              color: Colors.grey[400],
                            ),
                          ),
                        ],
                      ),
                    ),
                    TextButton(
                      onPressed: _disconnect,
                      child: const Text('Disconnect'),
                    ),
                  ],
                ),
              ),
            ] else ...[
              // URL field
              TextField(
                controller: _urlController,
                decoration: (widget.inputDecoration ?? const InputDecoration()).copyWith(
                  labelText: 'Supabase URL',
                  hintText: 'https://xxx.supabase.co',
                  prefixIcon: const Icon(Icons.link),
                ),
              ),
              const SizedBox(height: 12),

              // Anon Key field
              TextField(
                controller: _anonKeyController,
                obscureText: true,
                decoration: (widget.inputDecoration ?? const InputDecoration()).copyWith(
                  labelText: 'Anon Key',
                  hintText: 'eyJ...',
                  prefixIcon: const Icon(Icons.key),
                ),
              ),
              const SizedBox(height: 16),

              // Error message
              if (_error != null)
                Padding(
                  padding: const EdgeInsets.only(bottom: 12),
                  child: Text(
                    _error!,
                    style: const TextStyle(color: Colors.red, fontSize: 12),
                  ),
                ),

              // Save button
              SizedBox(
                width: double.infinity,
                child: ElevatedButton(
                  onPressed: _isSaving ? null : _saveCredentials,
                  style: ElevatedButton.styleFrom(
                    backgroundColor: widget.accentColor,
                  ),
                  child: _isSaving
                      ? const SizedBox(
                          height: 20,
                          width: 20,
                          child: CircularProgressIndicator(strokeWidth: 2),
                        )
                      : const Text('Connect'),
                ),
              ),

              // Help text
              Padding(
                padding: const EdgeInsets.only(top: 12),
                child: Text(
                  'Find these in your Supabase Dashboard under Settings > API',
                  style: TextStyle(
                    fontSize: 11,
                    color: Colors.grey[500],
                  ),
                ),
              ),
            ],
          ],
        );
      },
    );
  }

  String _getStatusText() {
    switch (_service.status) {
      case SupabaseConnectionStatus.notConfigured:
        return 'Not configured';
      case SupabaseConnectionStatus.connecting:
        return 'Connecting...';
      case SupabaseConnectionStatus.connected:
        return 'Connected';
      case SupabaseConnectionStatus.disconnected:
        return 'Disconnected';
      case SupabaseConnectionStatus.error:
        return 'Error';
    }
  }
}
